       var Game_Winner = (S) => 
       {
         let countA=0;
         let countD=0;
         for(let i=0; i<S.length; i++){
             if(S[i]==='A'){
                 countA+=1;
                }
             else{
                  countD+=1;
                }
            }
          if(countA>countD){
              return "Aditya";
            }

       var Palin_Check = (str) =>
       {
         let ans="";

              ans+=str[i];
            }
         if(ans===str){

            }
        };

       var Reverse_String = (str) => 
       {
